\section{Executive Summary}

\subsection{Problem Statement}

Aircraft maintenance is a critical operational function. Traditional reactive maintenance—fixing equipment only after it fails—leads to:
\begin{itemize}
    \item \textbf{Unexpected downtime} and flight cancellations
    \item \textbf{Emergency repairs} at significantly higher costs
    \item \textbf{Safety hazards} for passengers and crew
\end{itemize}

This project develops a machine learning-based predictive maintenance system to identify aircraft components likely to require maintenance within the next 10 flight cycles, enabling proactive scheduling.

\subsection{Solution Approach}

We implemented the \textbf{CRISP-DM} (Cross-Industry Standard Process for Data Mining) framework:
\begin{center}
\begin{tabularx}{0.8\textwidth}{|c|X|}
\toprule
\textbf{Phase} & \textbf{Objective} \\
\midrule
Business Understanding & Define success metrics prioritizing safety \\
Data Understanding & Analyze 6,000 maintenance records for patterns \\
Data Preparation & Handle 30:1 class imbalance with SMOTE \\
Modeling & Train and compare three candidate models \\
Evaluation & Comprehensive performance analysis \\
Deployment & Production-ready artifacts and API \\
\bottomrule
\end{tabularx}
\end{center}

\section{Phase 1: Business Understanding}

\subsection{Problem Definition}

\textbf{Core Question:} Can we predict aircraft component failures before they occur to enable proactive maintenance scheduling?

\textbf{Success Criteria:}
\begin{enumerate}
    \item \textcolor{critical}{\textbf{Recall > 95\%}} — Catch failures before they manifest (safety priority)
    \item \textbf{Precision > 60\%} — Minimize unnecessary maintenance (cost consideration)
    \item \textbf{ROC-AUC > 0.90} — Strong discrimination between failure and normal states
\end{enumerate}

\subsection{Decision Rationale: Safety-First Prioritization}

\begin{table}[H]
\centering
\begin{tabularx}{\textwidth}{|l|c|c|c|}
\hline
\textbf{Prediction Outcome} & \textbf{Actual Status} & \textbf{Cost} & \textbf{Priority} \\
\hline
\textbf{True Positive (TP)} & Predicted failure, actually fails & \$0 & \textcolor{success}{\checkmark} \\
\hline
\textbf{True Negative (TN)} & Predicted OK, actually OK & \$0 & \checkmark \\
\hline
\textbf{False Positive (FP)} & Predicted failure, actually OK & \$5K (wasted maintenance) & Acceptable \\
\hline
\textbf{False Negative (FN)} & Predicted OK, actually fails & \$500K+ (damage + safety) & \textcolor{critical}{CRITICAL} \\
\hline
\end{tabularx}
\caption{Cost-Benefit Analysis of Prediction Errors}
\end{table}

\textbf{Conclusion:} Minimize False Negatives (Maximize Recall) even at the cost of False Positives.

\subsection{Stakeholder Requirements}

\begin{description}
    \item[Maintenance Teams] Actionable alerts with clear maintenance windows and sensor readings
    \item[Operations] Minimum 5-7 day advance notice for scheduling
    \item[Finance] Cost-benefit analysis showing ROI within 12 months
    \item[Safety] Regulatory compliance and audit trails
\end{description}

\section{Phase 3: Data Preparation}

\subsection{Data Quality Issues Addressed}

\begin{enumerate}
    \item \textbf{Low-Variance Columns:} 
    \begin{itemize}
        \item Removed \texttt{last\_maintenance\_cycles} (all zeros — provides no information)
    \end{itemize}
    
    \item \textbf{Identifiers (non-predictive):}
    \begin{itemize}
        \item Removed \texttt{aircraft\_id}, \texttt{component\_id}
        \item These identify records but don't predict failures
    \end{itemize}
    
    \item \textbf{Result:} 11 core features retained for modeling
\end{enumerate}

\subsection{Handling Class Imbalance: SMOTE}

\subsubsection{The Problem}

A naive model trained on imbalanced data will simply predict "no failure" for all samples:
\begin{align*}
\text{Naive Accuracy} &= \frac{5807}{6000} = 96.78\% \quad \text{(useless but high!)} \\
\text{Naive Recall} &= \frac{0}{193} = 0\% \quad \text{(catches no failures — unacceptable)}
\end{align*}

\subsubsection{Our Solution: SMOTE}

\textbf{SMOTE} (Synthetic Minority Oversampling Technique) creates synthetic failure examples:

\begin{table}[H]
\centering
\begin{tabularx}{0.8\textwidth}{|l|c|c|c|}
\hline
& \textbf{Before SMOTE} & \textbf{After SMOTE} & \textbf{Change} \\
\hline
Class 0 (No Fail) & 4,646 & 4,646 & — \\
Class 1 (Fail) & 154 & 4,646 & $\uparrow 30.2\times$ \\
\hline
\textbf{Ratio} & 30.16:1 & 1:1 & Perfectly balanced \\
\hline
\end{tabularx}
\end{table}

\subsubsection{Why SMOTE vs. Alternatives?}

\begin{center}
\begin{tabularx}{\textwidth}{|l|c|c|c|}
\hline
\textbf{Method} & \textbf{Overfitting Risk} & \textbf{Information Loss} & \textbf{Effectiveness} \\
\hline
Random Oversampling & \textcolor{critical}{High} & None & Poor \\
Undersampling & None & \textcolor{critical}{High} & Poor \\
Class Weighting & Medium & None & Moderate \\
SMOTE (Selected) & \textcolor{success}{Low} & \textcolor{success}{None} & \textcolor{success}{Excellent} \\
\hline
\end{tabularx}
\end{center}

\textbf{Key Principle:} SMOTE applied \textbf{only to training data}, test set remains imbalanced for realistic evaluation.

\subsection{Feature Scaling}

\textbf{Method:} StandardScaler
\begin{equation}
x_{\text{scaled}} = \frac{x - \mu}{\sigma}
\end{equation}

\textbf{Why:} Logistic Regression requires normalized features for optimal performance.

\textbf{Data Leakage Prevention:} Scaler fitted on training data only, applied identically to test data.

\subsection{Train-Test Split}

\begin{itemize}
    \item \textbf{Ratio:} 80-20 stratified split
    \item \textbf{Stratification:} Preserves class distribution in both sets
    \item \textbf{Result:}
    \begin{itemize}
        \item Training: 4,800 samples (1,154 maintenance cases)
        \item Test: 1,200 samples (193 maintenance cases — realistic imbalance)
    \end{itemize}
\end{itemize}

\section{Phase 4: Modeling}

\subsection{Three-Model Comparison Philosophy}

Rather than relying on a single model, we evaluate three approaches:

\begin{description}
    \item[\textbf{Logistic Regression}]

\subsection{Model Selection: Logistic Regression}

\textbf{Rationale:}

\begin{enumerate}
    \item \textbf{Highest Recall (98.76\%):} Catches failures — our primary objective
    \item \textbf{Interpretability:} Coefficients directly explain feature importance
    \item \textbf{Production Safety:} Simpler models have fewer failure modes
    \item \textbf{Inference Speed:} Sub-millisecond predictions
    \item \textbf{Maintainability:} Easier for operations team to understand
\end{enumerate}

\textbf{Trade-off Accepted:} Slightly lower ROC-AUC than Random Forest, but superior recall and production characteristics.

\begin{enumerate}

    \item Re-run full CRISP-DM pipeline
    \item Compare new model vs. production model on holdout test set
    \item Deploy new model if performance improves (>1\% recall gain)
    \item Keep production model active if no improvement
\end{enumerate}

Monitor these metrics monthly:
\begin{itemize}

    \item Class imbalance ratio: Is failure rate changing over time?
    \item Model performance on recent data: Is recall degrading?
\end{itemize}

If drift detected exceeds thresholds, trigger immediate retraining.

\section{Key Trade-offs \& Design Decisions}

\begin{table}[H]

\begin{tabularx}{\textwidth}{|p{3cm}|X|X|}

& \textbf{Maximizing Recall (Our Choice)} & \textbf{Maximizing Precision} \\

\hline

\hline
Cost of Error & Catastrophic (crash, safety hazard) & Manageable (wasted maintenance) \\
\hline
Our Recall & 98.76\% & Could be 95\% with higher precision \\
\hline
Justification & \multicolumn{2}{l|}{In aviation, safety is paramount. Missing a failure is unacceptable.} \\

\subsection{Current Model Limitations}

\begin{enumerate}
    \item \textbf{Limited Minority Examples:} Only 193 failures in training set. More data would improve robustness.
    \item \textbf{Fixed 10-Cycle Window:} Model assumes maintenance needs occur within 10 cycles. Actual windows may vary.
    \item \textbf{Sensor Dependency:} Requires all 11 sensors operational. Missing sensors = no prediction.
    \item \textbf{Fleet Homogeneity:} Trained on specific aircraft type. New aircraft types need retraining.
    \item \textbf{Black-Box Warning:} While interpretable, Logistic Regression captures correlations not causation.
\end{enumerate}
