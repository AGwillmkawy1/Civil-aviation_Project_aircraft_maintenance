% SLIDE 7: SMOTE SOLUTION
% ============================================================================
\begin{frame}{Handling Imbalance: SMOTE}
\textbf{Problem:} 30:1 class imbalance breaks naive learning

\textbf{Solution:} SMOTE (Synthetic Minority Oversampling)

\vspace{0.3cm}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{Before SMOTE (Training):}
\begin{itemize}
    \item No Fail: 4,646 (96.8\%)
    \item Fail: 154 (3.2\%)
    \item Ratio: 30.16:1
\end{itemize}

\column{0.5\textwidth}
\textbf{After SMOTE (Training):}
\begin{itemize}
    \item No Fail: 4,646 (50\%)
    \item Fail: 4,646 (50\%)
    \item Ratio: 1:1 ✓
\end{itemize}
\end{columns}

\vspace{0.5cm}
\textbf{Key Principle:}
\begin{itemize}
    \item SMOTE creates \textcolor{accentblue}{synthetic examples} of the minority class
    \item Applied \textbf{only to training data} (model learns patterns)
    \item Test set remains imbalanced (realistic evaluation)
    \item \textcolor{success}{Result:} Model learns to detect failures without overfitting
\end{itemize}
\end{frame}

\textbf{Monthly Retraining Pipeline:}

\vspace{0.4cm}
\begin{enumerate}

    \item Combine with historical dataset
    \item Re-run CRISP-DM pipeline
